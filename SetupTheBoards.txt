Start by compiling OpenCV:
Here are the CMake option fixed to avoid compilation errors. Note : if you don't need one of the extra modules, you can remove the corresponding line, it will save you some time.

cmake -D CMAKE_BUILD_TYPE=RELEASE \
    -D CMAKE_INSTALL_PREFIX=/usr/local \
    -D INSTALL_C_EXAMPLES=OFF \
    -D INSTALL_PYTHON_EXAMPLES=OFF \
    -D OPENCV_GENERATE_PKGCONFIG=ON \
    -D ENABLE_NEON=ON \
    -D OPENCV_EXTRA_EXE_LINKER_FLAGS=-latomic \
    -D ENABLE_VFPV3=ON \
    -D BUILD_TESTS=OFF \
    -D OPENCV_ENABLE_NONFREE=ON \
    -D OPENCV_EXTRA_MODULES_PATH=/home/pi/Desktop/opencv_build/opencv_contrib/modules \
    -D WITH_1394=OFF \
    -D BUILD_EXAMPLES=OFF ..

For the instructions, you can follow this link, but you should use the previous options instead of the ones given here: 
https://linuxize.com/post/how-to-install-opencv-on-raspberry-pi/#install-opencv-from-source

Here are the commands:


sudo nano /etc/dphys-swapfile
Change the CONF_SWAPSIZE value from the default 100 to 1024
sudo apt update && sudo apt upgrade -y
sudo apt install build-essential cmake git pkg-config libgtk-3-dev "libcanberra-gtk*"
sudo apt install libavcodec-dev libavformat-dev libswscale-dev libv4l-dev libxvidcore-dev libx264-dev
sudo apt install libjpeg-dev libpng-dev libtiff-dev gfortran openexr libatlas-base-dev opencl-headers
sudo apt install python3-dev python3-numpy libtbb2 libtbb-dev libdc1394-22-dev
# change the following line with the path you want
mkdir ~/opencv_build && cd ~/opencv_build
git clone https://github.com/opencv/opencv.git
git clone https://github.com/opencv/opencv_contrib.git #Only if you want the contribs
#Same here, change the path to whichever you want
mkdir -p ~/opencv_build/opencv/build && cd ~/opencv_build/opencv/build

Paste the previous CMAKE -D command (From cmake to ..)
#-j4 depends of the core number of the board. The Pi has 4 cores, so use -j4
#On BBAI, use -j2
make -j4
sudo make install

#check if the install went well:
pkg-config --modversion opencv4
#should return 4.1.1 in my case. It will be higher for you, as it shows the current openCV version you installed.

sudo nano /etc/dphys-swapfile
CONF_SWAPSIZE=100
sudo /etc/init.d/dphys-swapfile restart

Install bcm2835 on RPi:

//(Replace X.XX with the most recent version (Found here http://www.open.com.au/mikem/bcm2835, beside page Title)
wget http://www.open.com.au/mikem/bcm2835/bcm2835-X.XX.tar.gz
tar xvfz bcm2835-1.5.tar.gz 
cd bcm2835-1.5
./configure
make
sudo make install

Install XRDP:
sudo apt-get install raspberrypi-ui-mods xinit xserver-xorg
sudo reboot
sudo apt install xrdp
systemctl show -p SubState --value xrdp
sudo adduser xrdp ssl-cert
#See https://linuxize.com/post/how-to-install-xrdp-on-raspberry-pi/
